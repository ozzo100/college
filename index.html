// تخزين الوثائق في localStorage let documents = JSON.parse(localStorage.getItem('documents')) || []; // عرض الأقسام function showSection(sectionId) { document.querySelectorAll('.main-content > div').forEach(div => { div.classList.add('hidden'); }); document.getElementById(sectionId + 'Section').classList.remove('hidden'); } // معالجة رفع الملف document.getElementById('fileInput').addEventListener('change', function(e) { const file = e.target.files[0]; if (file) { const reader = new FileReader(); reader.onload = function(e) { const newDoc = { name: file.name, type: file.type, size: (file.size / 1024).toFixed(2) + ' KB', date: new Date().toLocaleDateString(), data: e.target.result }; documents.push(newDoc); localStorage.setItem('documents', JSON.stringify(documents)); updateDocumentsTable(); }; reader.readAsDataURL(file); } }); // تحديث الجدول function updateDocumentsTable() { const tbody = document.getElementById('documentsList'); tbody.innerHTML = ''; documents.forEach((doc, index) => { tbody.innerHTML += ` ${doc.name} ${doc.type} ${doc.size} ${doc.date} `; }); } // عرض الوثيقة function viewDocument(index) { const doc = documents[index]; window.open(doc.data, '_blank'); } // حذف الوثيقة function deleteDocument(index) { documents.splice(index, 1); localStorage.setItem('documents', JSON.stringify(documents)); updateDocumentsTable(); } // تحديث الجدول عند تحميل الصفحة window.onload = updateDocumentsTable;
